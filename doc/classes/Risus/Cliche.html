unpack(text)

[Source]

# File risus.rb, line 144
    def Cliche.unpack(text)
      begin
        name, value, is_double, funky = text.split(/:/)
      rescue Exception
        raise RisusError, "Trouble unpacking Cliche: #{text}"
      end
      return [name, value, is_double.to_i.zero? ? false : true, funky.to_sym()]
    end
